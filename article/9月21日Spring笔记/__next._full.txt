        System.out.println(<span class="hljs-string">"userId="</span> + userId);
        System.out.println(<span class="hljs-string">"userName="</span> + userName);
        <span class="hljs-keyword">return</span> <span class="hljs-string">"main"</span>;
    }
</code></pre>
<h3 id="cookievalue获取cookie值"><a href="#cookievalue获取cookie值"><code>@CookieValue</code>获取cookie值</a></h3>

        System.out.println(<span class="hljs-string">"userId="</span> + userId);

        System.out.println(cookieValue);
        <span class="hljs-keyword">return</span> <span class="hljs-string">"main"</span>;
    }
</code></pre>
<h3 id="headervalue获取header值"><a href="#headervalue获取header值"><code>@HeaderValue</code>获取header值</a></h3>
