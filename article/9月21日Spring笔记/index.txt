<h3 id="requestparam自定义传参"><a href="#requestparam自定义传参"><code>@RequestParam</code>自定义传参</a></h3>

<pre><code class="hljs language-java">    <span class="hljs-meta">@RequestMapping("/value.action")</span>

        System.out.println(<span class="hljs-string">"userId="</span> + userId);
        System.out.println(<span class="hljs-string">"userName="</span> + userName);
        <span class="hljs-keyword">return</span> <span class="hljs-string">"main"</span>;
    }
</code></pre>
<h3 id="cookievalue获取cookie值"><a href="#cookievalue获取cookie值"><code>@CookieValue</code>获取cookie值</a></h3>

<pre><code class="hljs language-java">    <span class="hljs-meta">@RequestMapping("/value.action")</span>
    <span class="hljs-keyword">public</span> String <span class="hljs-title function_">value</span><span class="hljs-params">(<span class="hljs-meta">@RequestParam("id")</span> String userId, <span class="hljs-meta">@RequestParam("name")</span> String userName, <span class="hljs-meta">@CookieValue("JSESSIONID")</span> String cookieValue)</span>{
        System.out.println(<span class="hljs-string">"userId="</span> + userId);
        System.out.println(<span class="hljs-string">"userName="</span> + userName);
        System.out.println(cookieValue);
        <span class="hljs-keyword">return</span> <span class="hljs-string">"main"</span>;
    }
</code></pre>
<h3 id="headervalue获取header值"><a href="#headervalue获取header值"><code>@HeaderValue</code>获取header值</a></h3>

<pre><code class="hljs language-java">    <span class="hljs-meta">@RequestMapping("/value.action")</span>
    <span class="hljs-keyword">public</span> String <span class="hljs-title function_">value</span><span class="hljs-params">(<span class="hljs-meta">@RequestParam("id")</span> String userId, <span class="hljs-meta">@RequestParam("name")</span> String userName, <span class="hljs-meta">@CookieValue("JSESSIONID")</span> String cookieValue, <span class="hljs-meta">@RequestHeader("User-Agent")</span>String headerValue)</span>{
        System.out.println(<span class="hljs-string">"userId="</span> + userId);

        System.out.println(cookieValue);
        System.out.println(headerValue);
        <span class="hljs-keyword">return</span> <span class="hljs-string">"main"</span>;
    }
