Promiseзӣёе…іеҮҪж•°зІҫз®Җе®һзҺ°

2026е№ҙ2жңҲ27ж—ҘВ·618 еӯ—В·зәҰ 3 еҲҶй’ҹ
Obsidian

зҗҶи§Ј Promise зҡ„ж ёеҝғеңЁдәҺзҗҶи§Је®ғжҳҜдёҖдёӘзҠ¶жҖҒжңәе’Ңеӣһи°ғйҳҹеҲ—зҡ„йӣҶеҗҲгҖӮдёәдәҶиҫҫеҲ°вҖңеҺ»зІ—еҸ–зІҫвҖқзҡ„ж•ҲжһңпјҢжҲ‘д»¬дёҚйңҖиҰҒе®һзҺ°е®Ңж•ҙзҡ„ Promises/A+ и§„иҢғпјҲйӮЈиҝҮдәҺеҶ—й•ҝпјүпјҢиҖҢжҳҜжҠ“дҪҸе…¶е№¶еҸ‘жҺ§еҲ¶е’ҢзҠ¶жҖҒжөҒиҪ¬зҡ„жң¬иҙЁйҖ»иҫ‘гҖӮ

д»ҘдёӢжҳҜ Promise жңҖж ёеҝғзҡ„еҮ дёӘйқҷжҖҒж–№жі•еҸҠжһ„йҖ йҖ»иҫ‘зҡ„зІҫз®Җе®һзҺ°пјҡ

1. ж ёеҝғжңәеҲ¶пјҡPromise.all

иҝҷжҳҜйқўиҜ•е’Ңе®һжҲҳдёӯжңҖеёёи§Ғзҡ„йҖ»иҫ‘гҖӮе®ғзҡ„зІҫй«“еңЁдәҺпјҡе№¶иЎҢжү§иЎҢпјҢдёІиЎҢзӯүеҫ…пјҢдёҖй”ҷе…Ёй”ҷгҖӮ

JavaScript

Promise.myAll = function(promises) {
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;
    const len = promises.length;

    if (len === 0) return resolve([]);

    promises.forEach((p, i) => {
      // еҢ…иЈ…жҲҗ Promise д»ҘеӨ„зҗҶйқһ Promise еҖј
      Promise.resolve(p).then(
        res => {
          results[i] = res; // дҝқиҜҒз»“жһңйЎәеәҸ
          count++;
          if (count === len) resolve(results);
        },
        err => reject(err) // еҸӘиҰҒжңүдёҖдёӘеӨұиҙҘпјҢж•ҙдҪ“з«ӢеҚіеӨұиҙҘ
      );
    });
  });
};

2. з«һйҖҹжңәеҲ¶пјҡPromise.race

е®ғзҡ„йҖ»иҫ‘жңҖзәҜзІ№пјҡи°Ғе…Ҳе®ҢжҲҗпјҲж— и®әжҲҗеҠҹеӨұиҙҘпјүпјҢе°ұз”Ёи°Ғзҡ„з»“жһңгҖӮ

JavaScript

Promise.myRace = function(promises) {
  return new Promise((resolve, reject) => {
    promises.forEach(p => {
      // з¬¬дёҖдёӘж”№еҸҳзҠ¶жҖҒзҡ„ Promise дјҡзӣҙжҺҘи§ҰеҸ‘еӨ–еұӮ Promise зҡ„ resolve/reject
      Promise.resolve(p).then(resolve, reject);
    });
  });
};

3. е®№й”ҷжңәеҲ¶пјҡPromise.allSettled

дёҺ all дёҚеҗҢпјҢе®ғдёҚе…іеҝғиғңиҙҹпјҢеҸӘе…іеҝғвҖңжҳҜеҗҰз»“жқҹвҖқгҖӮиҝҷеңЁйңҖиҰҒиҺ·еҸ–жүҖжңүејӮжӯҘз»“жһңпјҲж— и®әеҘҪеқҸпјүж—¶йқһеёёжңүз”ЁгҖӮ

JavaScript

Promise.myAllSettled = function(promises) {
  return Promise.all(
    promises.map(p => 
      Promise.resolve(p).then(
        value => ({ status: 'fulfilled', value }),
        reason => ({ status: 'rejected', reason })
      )
    )
  );
};

4. еҒҘеЈ®жҖ§пјҡPromise.any

иҝҷжҳҜвҖңе…ЁиҙҘжүҚиҙҘпјҢдёҖиғңеҚіиғңвҖқзҡ„йҖ»иҫ‘гҖӮ

JavaScript

Promise.myAny = function(promises) {
  return new Promise((resolve, reject) => {
    let count = 0;
    const errors = [];
    const len = promises.length;

    if (len === 0) return reject(new AggregateError('All promises were rejected'));

    promises.forEach((p, i) => {
      Promise.resolve(p).then(
        res => resolve(res), // еҸӘиҰҒжңүдёҖдёӘжҲҗеҠҹпјҢзӣҙжҺҘиҝ”еӣһ
        err => {
          errors[i] = err;
          count++;
          if (count === len) reject(new AggregateError(errors));
        }
      );
    });
  });
};

5. зҠ¶жҖҒжңәжң¬иҙЁпјҡз®Җжҳ“жһ„йҖ еҷЁ

еҰӮжһңиҰҒзҗҶи§Ј Promise еҶ…йғЁжҳҜеҰӮдҪ•е·ҘдҪңзҡ„пјҢзңӢиҝҷж®өжһҒз®Җзҡ„жһ„йҖ йҖ»иҫ‘еҚіеҸҜпјҡ

JavaScript

function MyPromise(executor) {
  this.state = 'pending';
  this.value = null;
  this.callbacks = []; // еӯҳеӮЁ then йҮҢзҡ„еӣһи°ғ

  const resolve = (val) => {
    if (this.state !== 'pending') return;
    this.state = 'fulfilled';
    this.value = val;
    this.callbacks.forEach(cb => cb.onFulfilled(val));
  };

  const reject = (reason) => {
    if (this.state !== 'pending') return;
    this.state = 'rejected';
    this.value = reason;
    this.callbacks.forEach(cb => cb.onRejected(reason));
  };

  try {
    executor(resolve, reject);
  } catch (e) {
    reject(e);
  }
}

жҖ»з»“

  • all: жүҖжңүдәәеҝ…йЎ»жҲҗеҠҹпјҢдёҖдёӘеӨұиҙҘе°ұе®ҢиӣӢгҖӮ

  • race: еҸӘзңӢз¬¬дёҖеҗҚпјҢдёҚй—®еҜ№й”ҷгҖӮ

  • allSettled: жүҖжңүдәәжҠҘдёӘе№іе®үпјҲж— и®әжӯ»жҙ»пјүгҖӮ

  • any: еҸӘиҰҒжңүдёҖдёӘжҙ»зқҖзҡ„е°ұиЎҢпјҢе…Ёжӯ»дәҶжүҚж”ҫејғгҖӮ

иҝҷдәӣйҖ»иҫ‘зҡ„е…ұжҖ§еңЁдәҺйҖҡиҝҮй—ӯеҢ…з»ҙжҠӨи®Ўж•°еҷЁпјҲCounterпјүжҲ–иҖ…зӣҙжҺҘз«һдәүзҠ¶жҖҒж§ҪгҖӮжҺҢжҸЎдәҶиҝҷеҮ дёӘпјҢе°ұжҺҢжҸЎдәҶејӮжӯҘжөҒжҺ§еҲ¶зҡ„жң¬иҙЁгҖӮ

еҰӮжһңдҪ еҜ№жҹҗдёӘзү№е®ҡзҡ„е№¶еҸ‘жҺ§еҲ¶пјҲжҜ”еҰӮеёҰе№¶еҸ‘йҷҗеҲ¶зҡ„ Promise и°ғеәҰеҷЁпјүж„ҹе…ҙи¶ЈпјҢжҲ‘еҸҜд»Ҙз»§з»ӯдёәдҪ и§Јжһ„гҖӮдҪ жғідәҶи§ЈеҰӮдҪ•жҺ§еҲ¶ејӮжӯҘд»»еҠЎзҡ„е№¶еҸ‘йў‘зҺҮеҗ—пјҹ
