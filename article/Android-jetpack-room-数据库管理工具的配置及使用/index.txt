    <h2>配置</h2>
<p>在根目录的<code>build.gradle</code>的<code>plugin</code>下添加</p>
<pre><code class="language-gradle">id 'com.google.devtools.ksp' version '1.8.21-1.0.11' apply false
</code></pre>
<p>注意，版本号<code>1.8.21</code>要与<code>org.jetbrains.kotlin.android</code>版本号一致。示例:</p>
<p><code>/build.gradle</code></p>
<pre><code class="language-gradle">plugins {
    id 'com.android.application' version '8.0.1' apply false
    id 'com.android.library' version '8.0.1' apply false
    id 'org.jetbrains.kotlin.android' version '1.8.21' apply false
    id 'com.google.devtools.ksp' version '1.8.21-1.0.11' apply false
}
</code></pre>
<p>在<code>app/build.gradle</code>下添加</p>
<pre><code class="language-gradle"> def room_version = "2.5.0"

    implementation "androidx.room:room-runtime:$room_version"
    annotationProcessor "androidx.room:room-compiler:$room_version"

    // To use Kotlin Symbol Processing (KSP)
    ksp "androidx.room:room-compiler:$room_version"


    // optional - Test helpers
    testImplementation "androidx.room:room-testing:$room_version"

<h2>使用</h2>
<h3>创建数据库抽象类</h3>
<p><code>AppDataBase.kt</code></p>
<pre><code class="language-kotlin">@Database(entities = [User::class], version = 1)
abstract class AppDatabase : RoomDatabase(){
    abstract fun userDao(): UserDao
}
</code></pre>
<h3>创建entity类</h3>
<p><code>User.kt</code></p>
<pre><code class="language-kotlin">@Entity
data class User (
    @PrimaryKey val uid: Int,
    @ColumnInfo(name = "name") val name: String?,
    @ColumnInfo(name = "sex") val sex: String?
)
</code></pre>
<h3>创建Dao类</h3>
<pre><code class="language-kotlin">package com.xyl.roomdemo.dao

import androidx.room.Dao
import androidx.room.Delete
import androidx.room.Insert
import androidx.room.Query
import androidx.room.Update
import com.xyl.roomdemo.entity.User

@Dao
interface UserDao {
    @Query("select * from user")
    fun getAll(): List&#x3C;User>
    @Query("select * from user where uid = (:queryId)")
    fun getUserById(queryId: Int): User
    @Insert
    fun addUser(user: User)
    @Update
    fun updateUser(user: User)
    @Delete
    fun delUser(user: User)

}
</code></pre>
<h3>配置数据库</h3>
<p><code>App.kt</code></p>
<pre><code class="language-kotlin">class App: Application() {
    lateinit var db: AppDatabase
    override fun onCreate() {
        super.onCreate()
        Log.i(TAG, "onCreate: app启动")
        GlobalScope.launch(Dispatchers.IO) {
            db = Room.databaseBuilder(applicationContext, AppDatabase::class.java, "database-name")
                .build()
        }
    }
}
</code></pre>
<h3>使用示例</h3>
<p><code>MainActivity.kt</code></p>
<pre><code class="language-kotlin">package com.xyl.roomdemo

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.util.Log
import com.xyl.roomdemo.entity.User
import com.xyl.roomdemo.utils.TAG
import kotlinx.coroutines.DelicateCoroutinesApi
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.GlobalScope
import kotlinx.coroutines.launch
import kotlinx.coroutines.withContext

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        val app: App = application as App

        GlobalScope.launch(Dispatchers.IO) {
            val dao = app.db.userDao()
            dao.addUser(User(1001, "小明", "男"))
            dao.addUser(User(1002, "小红", "女"))
            dao.addUser(User(1003, "小花", "女"))
            dao.addUser(User(1004, "张三", "男"))
            dao.delUser(User(1002, "", ""))
            val userList = dao.getAll()
            withContext(Dispatchers.Main) {
                Log.i(TAG, "onCreate: ${userList.toString()}")
            }

        }

    }
}
</code></pre>
